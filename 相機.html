import cv2 import numpy as np import mediapipe as mp def create_lut(whiten_level): """建立美白 LUT""" lut = np.zeros(256, dtype=np.uint8) for i in range(256): val = i * whiten_level lut[i] = np.clip(val, 0, 255) return lut def get_face_mask(image, face_mesh): """產生臉部 mask""" h, w = image.shape[:2] img_rgb = cv2.cvtColor(image, cv2.COLOR_BGR2RGB) results = face_mesh.process(img_rgb) mask = np.zeros((h, w), dtype=np.uint8) if results.multi_face_landmarks: for face_landmarks in results.multi_face_landmarks: points = [] for lm in face_landmarks.landmark: x = int(lm.x * w) y = int(lm.y * h) points.append((x, y)) points = np.array(points, dtype=np.int32) hull = cv2.convexHull(points) cv2.fillConvexPoly(mask, hull, 255) # 擴展到脖子 + 柔化邊緣 mask = cv2.GaussianBlur(mask, (31, 31), 0) return mask def ai_beauty_whitening(image_path, whiten_level=1.5): img = cv2.imread(image_path) if img is None: print("❌ 圖片讀取失敗") return None h, w = img.shape[:2] mp_face_mesh = mp.solutions.face_mesh face_mesh = mp_face_mesh.FaceMesh( static_image_mode=True, max_num_faces=1, refine_landmarks=True ) # ===== 1. Face Mask ===== mask = get_face_mask(img, face_mesh) mask = mask.astype(np.float32) / 255.0 mask = cv2.cvtColor(mask, cv2.COLOR_GRAY2BGR) # ===== 2. LUT 美白 ===== lut = create_lut(whiten_level) whiten_img = cv2.LUT(img, lut) # ===== 3. 磨皮（bilateral）===== smooth_img = cv2.bilateralFilter( whiten_img, d=7, sigmaColor=50, sigmaSpace=50 ) # ===== 4. 融合（關鍵修正）===== mask = mask.astype(np.float32) img_f = img.astype(np.float32) smooth_f = smooth_img.astype(np.float32) output = img_f * (1 - mask) + smooth_f * mask output = np.clip(output, 0, 255).astype(np.uint8) face_mesh.close() return output # ========================= # 🔥 測試 # ========================= if __name__ == "__main__": input_file = "input.jpg" result = ai_beauty_whitening( input_file, whiten_level=1.8 ) if result is not None: cv2.imshow("Original", cv2.imread(input_file)) cv2.imshow("AI Beauty Fixed", result) cv2.imwrite("ai_beauty_result.jpg", result) print("✅ 完成！已輸出 ai_beauty_result.jpg") cv2.waitKey(0) cv2.destroyAllWindows()
