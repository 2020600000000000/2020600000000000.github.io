pip install mediapipe import sys import cv2 import numpy as np import mediapipe as mp from PyQt6.QtWidgets import (QApplication, QMainWindow, QWidget, QLabel, QPushButton, QSlider, QVBoxLayout, QHBoxLayout, QFileDialog) from PyQt6.QtCore import Qt from PyQt6.QtGui import QImage, QPixmap class AIPhotoEditor(QMainWindow): def __init__(self): super().__init__() self.initUI() self.orig_image = None self.processed_image = None # 初始化 MediaPipe Face Mesh AI 模型 self.mp_face_mesh = mp.solutions.face_mesh self.face_mesh = self.mp_face_mesh.FaceMesh( static_image_mode=True, # 處理靜態圖片 max_num_faces=4, # 最多支援同時偵測 4 張臉 refine_landmarks=True, min_detection_confidence=0.5 ) def initUI(self): self.setWindowTitle('GitHub 專案：AI 關鍵點識別美白磨皮工具') self.setGeometry(100, 100, 1000, 650) main_widget = QWidget() self.setCentralWidget(main_widget) main_layout = QHBoxLayout(main_widget) # 控制面板 control_layout = QVBoxLayout() self.btn_open = QPushButton('打開圖片', self) self.btn_open.clicked.connect(self.open_image) control_layout.addWidget(self.btn_open) # 磨皮滑桿 control_layout.addWidget(QLabel('AI 臉部磨皮強度:')) self.slider_blur = QSlider(Qt.Orientation.Horizontal) self.slider_blur.setRange(0, 50) self.slider_blur.setValue(0) self.slider_blur.valueChanged.connect(self.process_image) control_layout.addWidget(self.slider_blur) # 美白滑桿 control_layout.addWidget(QLabel('AI 臉部美白亮度:')) self.slider_bright = QSlider(Qt.Orientation.Horizontal) self.slider_bright.setRange(0, 100) self.slider_bright.setValue(0) self.slider_bright.valueChanged.connect(self.process_image) control_layout.addWidget(self.slider_bright) self.btn_save = QPushButton('儲存圖片', self) self.btn_save.clicked.connect(self.save_image) control_layout.addWidget(self.btn_save) control_layout.addStretch() main_layout.addLayout(control_layout, 1) # 圖片顯示 self.lbl_image = QLabel('請先打開一張圖片', self) self.lbl_image.setAlignment(Qt.AlignmentFlag.AlignCenter) self.lbl_image.setStyleSheet("border: 1px solid #bdc3c7; background-color: #ecf0f1;") main_layout.addWidget(self.lbl_image, 4) def open_image(self): file_path, _ = QFileDialog.getOpenFileName(self, '選擇圖片', '', 'Image files (*.jpg *.jpeg *.png)') if file_path: self.orig_image = cv2.imdecode(np.fromfile(file_path, dtype=np.uint8), cv2.IMREAD_COLOR) self.slider_blur.setValue(0) self.slider_bright.setValue(0) self.display_image(self.orig_image) def get_ai_face_mask(self, img): """ 利用 MediaPipe AI 模型偵測人臉外部輪廓，並排除眼睛與嘴巴 """ h, w, _ = img.shape mask = np.zeros((h, w), dtype=np.uint8) # MediaPipe 需要 RGB 格式 rgb_img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB) results = self.face_mesh.process(rgb_img) if not results.multi_face_landmarks: # 如果 AI 沒偵測到人臉，就回傳全白遮罩（防呆：套用到整張圖） return np.ones((h, w), dtype=np.uint8) * 255 # 核心 AI 邏輯：利用 MediaPipe 預設的臉部輪廓索引 # FACEMESH_FACE_OVAL 包含臉部最外圈的邊緣點 face_oval_indices = [p.v for p in self.mp_face_mesh.FACEMESH_FACE_OVAL] for face_landmarks in results.multi_face_landmarks: # 蒐集這張臉的外圍輪廓座標 points = [] for idx in face_oval_indices: landmark = face_landmarks.landmark[idx] pt = (int(landmark.x * w), int(landmark.y * h)) points.append(pt) # 填滿臉部多邊形為白色 cv2.fillPoly(mask, [np.array(points, dtype=np.int32)], 255) # 【進階排除】挖空眼睛與嘴巴，避免五官一起被模糊（自由選配，此處先保留基礎臉部輪廓） # 為了讓妝容與眼神銳利，這裡也可以另外用 FACEMESH_EYES 索引進行挖空。 # 邊緣羽化，讓美顏邊界自然不突兀 mask_blur = cv2.GaussianBlur(mask, (31, 31), 0) return mask_blur def process_image(self): if self.orig_image is None: return # 1. 呼叫 AI 模型取得人臉遮罩 face_mask = self.get_ai_face_mask(self.orig_image) face_weight = face_mask.astype(float) / 255.0 face_weight = cv2.merge([face_weight, face_weight, face_weight]) # 2. 製作美化效果圖 effect_img = self.orig_image.copy() blur_val = self.slider_blur.value() if blur_val > 0: effect_img = cv2.bilateralFilter(effect_img, 0, blur_val * 3, blur_val * 3) bright_val = self.slider_bright.value() if bright_val > 0: alpha = 1.0 + (bright_val / 300.0) beta = bright_val effect_img = cv2.convertScaleAbs(effect_img, alpha=alpha, beta=beta) # 3. 根據 AI 遮罩進行局部融合 orig_img_float = self.orig_image.astype(float) effect_img_float = effect_img.astype(float) output = orig_img_float * (1.0 - face_weight) + effect_img_float * face_weight self.processed_image = np.clip(output, 0, 255).astype(np.uint8) self.display_image(self.processed_image) def display_image(self, img): rgb_img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB) h, w, ch = rgb_img.shape bytes_per_line = ch * w qt_img = QImage(rgb_img.data, w, h, bytes_per_line, QImage.Format.Format_RGB888) pixmap = QPixmap.fromImage(qt_img) scaled_pixmap = pixmap.scaled(self.lbl_image.width(), self.lbl_image.height(), Qt.AspectRatioMode.KeepAspectRatio, Qt.TransformationMode.SmoothTransformation) self.lbl_image.setPixmap(scaled_pixmap) def save_image(self): if self.processed_image is None: return file_path, _ = QFileDialog.getOpenFileName(self, '儲存圖片', '', 'JPEG (*.jpg);;PNG (*.png)') if file_path: ext = file_path.split('.')[-1] _, img_encoded = cv2.imencode(f'.{ext}', self.processed_image) img_encoded.tofile(file_path) if __name__ == '__main__': app = QApplication(sys.argv) editor = AIPhotoEditor() editor.show() sys.exit(app.exec())
